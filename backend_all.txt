@router.post("/register", response_model=UserResponse)

    logger.info(f"рџ”Ќ REGISTER ATTEMPT: {user_data.email}")

    try:
        result = await db.execute(select(User).where(User.email == user_data.email))
        existing = result.scalar_one_or_none()
        if existing:
            logger.warning(f"вќЊ Email already registered: {user_data.email}")
            raise HTTPException(status_code=400, detail="Email already registered")

        user = User(
            email=user_data.email,
            password_hash=hash_password(user_data.password)
        )
        db.add(user)
        await db.commit()
        await db.refresh(user)

        logger.info(f"вњ… User created: {user.id} - {user.email}")

        return UserResponse(
            id=user.id,
            email=user.email,
            created_at=user.created_at

        raise
    except Exception as e:
        logger.error(f"рџ”Ґ REGISTER ERROR: {type(e).__name__} - {str(e)}")
        await db.rollback()
        raise HTTPException(status_code=500, detail=f"Internal error: {str(e)}")

                    book_id=existing_book.id,

                )

        await db.flush()

        user_book = UserBook(
            user_id=current_user.id,
            book_id=new_book.id,
            status="planned"
        )
        db.add(user_book)

        await db.commit()
        await db.refresh(new_book)

        return new_book

async def get_current_user(
    credentials: Annotated[HTTPAuthorizationCredentials, Depends(security)],
    db: Annotated[AsyncSession, Depends(get_db)]
) -> User:
    """РџРѕР»СѓС‡РёС‚СЊ С‚РµРєСѓС‰РµРіРѕ РїРѕР»СЊР·РѕРІР°С‚РµР»СЏ РїРѕ JWT С‚РѕРєРµРЅСѓ (РІСЂРµРјРµРЅРЅРѕ СѓРїСЂРѕС‰РµРЅРѕ)"""
    token = credentials.credentials
    payload = decode_token(token)
    
    if payload is None:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Invalid token",
            headers={"WWW-Authenticate": "Bearer"},
        )
    
    user_id = payload.get("sub")
    if user_id is None:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Invalid token payload",
        )
    
    # Р’СЂРµРјРµРЅРЅРѕ: РІРѕР·РІСЂР°С‰Р°РµРј РїРµСЂРІРѕРіРѕ РїРѕР»СЊР·РѕРІР°С‚РµР»СЏ (С‚РѕР»СЊРєРѕ РґР»СЏ С‚РµСЃС‚Р°!)
    # РџСЂРµРѕР±СЂР°Р·СѓРµРј СЃС‚СЂРѕРєСѓ РІ UUID, РµСЃР»Рё РЅСѓР¶РЅРѕ
    if isinstance(user_id, str):
        try:
            user_id = UUID(user_id)
        except ValueError:
            # Р•СЃР»Рё РЅРµ UUID, РёС‰РµРј РїРѕ email РёР»Рё СЃРѕР·РґР°С‘Рј С„РµР№РєРѕРІРѕРіРѕ
            result = await db.execute(select(User).limit(1))
            user = result.scalar_one_or_none()
            if user:
                return user
            raise HTTPException(
                status_code=status.HTTP_401_UNAUTHORIZED,
                detail="Invalid user ID format",
            )
    
    result = await db.execute(select(User).where(User.id == user_id))
    user = result.scalar_one_or_none()
    
    if user is None:
        # Р’СЂРµРјРµРЅРЅРѕ: РµСЃР»Рё РїРѕР»СЊР·РѕРІР°С‚РµР»СЊ РЅРµ РЅР°Р№РґРµРЅ, РІРѕР·РІСЂР°С‰Р°РµРј РїРµСЂРІРѕРіРѕ (С‚РѕР»СЊРєРѕ РґР»СЏ С‚РµСЃС‚Р°!)
        result = await db.execute(select(User).limit(1))
        user = result.scalar_one_or_none()
        if user:
            return user
        
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="User not found",
        )
    
    return user
